DEPLOYMENT COMMANDS
───────────────────
# 1. Commit code
git add .
git commit -F .git-commit-message.txt

# 2. Push to GitHub
git remote add origin https://github.com/clockwork-innovations/simply-mcp-ts.git
git branch -M main
git push -u origin main

# 3. Tag release
git tag -a v1.0.0 -m "Release v1.0.0"
git push origin v1.0.0

# 4. Publish to npm
npm login
npm publish --access public  # if using scoped name
# or
npm publish  # if using simply-mcp

VERIFY DEPLOYMENT
─────────────────
# Check GitHub
Visit: https://github.com/clockwork-innovations/simply-mcp-ts

# Check npm
npm view simply-mcp

# Test install
npm install simply-mcp
